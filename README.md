    <script>
        // Three.js Scene Setup
        let scene, camera, renderer, carGroup, controls;
        let currentColor = 0xFF0000;
        let currentWheel = 0;
        let currentSpoiler = 0;
        let currentEngine = 'stock';
        let basePrice = 40000;
        let totalPrice = 40000;

        const colors = [
            0xFF0000, 0x00FF00, 0x0000FF, 0xFFFF00, 0xFF00FF, 0x00FFFF,
            0xFF6B6B, 0x4ECDC4, 0x45B7D1, 0x96CEB4, 0xFFEAA7, 0xDDA0DD
        ];

        const wheelStyles = ['Classic', 'Sport', 'Racing', 'Chrome', 'Matte Black'];
        const spoilerStyles = ['None', 'Subtle', 'Aggressive', 'GT Wing'];

        init();
        animate();

        function init() {
            // Scene
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0x222222);

            // Camera
            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(5, 3, 5);

            // Renderer
            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth * 0.65, window.innerHeight);
            renderer.shadowMap.enabled = true;
            renderer.shadowMap.type = THREE.PCFSoftShadowMap;
            document.getElementById('carViewer').appendChild(renderer.domElement);

            // Controls
            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;

            // Lighting
            const ambientLight = new THREE.AmbientLight(0x404040, 0.6);
            scene.add(ambientLight);

            const directionalLight = new THREE.DirectionalLight(0xffffff, 0.8);
            directionalLight.position.set(10, 10, 5);
            directionalLight.castShadow = true;
            directionalLight.shadow.mapSize.width = 2048;
            directionalLight.shadow.mapSize.height = 2048;
            scene.add(directionalLight);

            // Ground
            const groundGeometry = new THREE.PlaneGeometry(50, 50);
            const groundMaterial = new THREE.MeshLambertMaterial({ color: 0x333333 });
            const ground = new THREE.Mesh(groundGeometry, groundMaterial);
            ground.rotation.x = -Math.PI / 2;
            ground.receiveShadow = true;
            scene.add(ground);

            // Create Car
            createCar();

            // Setup UI
            setupUI();
        }

        function createCar() {
            carGroup = new THREE.Group();

            // Car Body
            const bodyGeometry = new THREE.BoxGeometry(4, 1.5, 2);
            const bodyMaterial = new THREE.MeshPhongMaterial({ color: currentColor });
            const body = new THREE.Mesh(bodyGeometry, bodyMaterial);
            body.position.y = 1;
            body.castShadow = true;
            carGroup.add(body);

            // Car Roof
            const roofGeometry = new THREE.BoxGeometry(2.5, 0.8, 1.8);
            const roofMaterial = new THREE.MeshPhongMaterial({ color: currentColor });
            const roof = new THREE.Mesh(roofGeometry, roofMaterial);
            roof.position.set(0, 2.2, 0);
            roof.castShadow = true;
            carGroup.add(roof);

            // Wheels
            createWheels();

            // Spoiler
            createSpoiler();

            carGroup.castShadow = true;
            scene.add(carGroup);
        }

        function createWheels() {
            // Clear existing wheels
            carGroup.children.forEach(child => {
                if (child.userData.isWheel) {
                    carGroup.remove(child);
                }
            });

            const wheelGeometry = new THREE.CylinderGeometry(0.4, 0.4, 0.3, 16);
            const wheelMaterial = new THREE.MeshPhongMaterial({ color: 0x333333 });

            const wheelPositions = [
                [-1.5, 0.5, 1.2], [1.5, 0.5, 1.2],
                [-1.5, 0.5, -1.2], [1.5, 0.5, -1.2]
            ];

            wheelPositions.forEach(pos => {
                const wheel = new THREE.Mesh(wheelGeometry, wheelMaterial);
                wheel.rotation.z = Math.PI / 2;
                wheel.position.set(...pos);
                wheel.castShadow = true;
                wheel.userData.isWheel = true;
                carGroup.add(wheel);
            });
        }

        function createSpoiler() {
            // Clear existing spoiler
            carGroup.children.forEach(child => {
                if (child.userData.isSpoiler) {
                    carGroup.remove(child);
                }
            });

            if (currentSpoiler === 0) return;

            const spoilerGeometry = new THREE.BoxGeometry(2.5, 0.2, 0.3);
            const spoilerMaterial = new THREE.MeshPhongMaterial({ color: currentColor });
            const spoiler = new THREE.Mesh(spoilerGeometry, spoilerMaterial);
            spoiler.position.set(0, 2.8, -1.3);
            spoiler.castShadow = true;
            spoiler.userData.isSpoiler = true;
            carGroup.add(spoiler);
        }

        function setupUI() {
            // Color picker
            const colorPicker = document.getElementById('colorPicker');
            colors.forEach((color, index) => {
                const colorOption = document.createElement('div');
                colorOption.className = 'color-option';
                colorOption.style.backgroundColor = `#${color.toString(16).padStart(6, '0')}`;
                if (index === 0) colorOption.classList.add('active');
                colorOption.onclick = () => selectColor(index);
                colorPicker.appendChild(colorOption);
            });

            // Wheel options
            const wheelOptions = document.getElementById('wheelOptions');
            wheelStyles.forEach((style, index) => {
                const btn = document.createElement('button');
                btn.className = 'wheel-btn';
                btn.textContent = style;
                if (index === 0) btn.classList.add('active');
                btn.onclick = () => selectWheel(index);
                wheelOptions.appendChild(btn);
            });

            // Spoiler options
            const spoilerOptions = document.getElementById('spoilerOptions');
            spoilerStyles.forEach((style, index) => {
                const btn = document.createElement('button');
                btn.className = 'spoiler-btn';
                btn.textContent = style;
                if (index === 0) btn.classList.add('active');
                btn.onclick = () => selectSpoiler(index);
                spoilerOptions.appendChild(btn);
            });
        }

        function selectColor(index) {
            currentColor = colors[index];
            
            // Update car body color
            carGroup.children.forEach(child => {
                if (!child.userData.isWheel && !child.userData.isSpoiler) {
                    child.material.color.setHex(currentColor);
                }
            });

            // Update UI
            document.querySelectorAll('.color-option').forEach((el, i) => {
                el.classList.toggle('active', i === index);
            });

            updatePrice();
        }

        function selectWheel(index) {
            currentWheel = index;
            createWheels();
            
            document.querySelectorAll('.wheel-btn').forEach((el, i) => {
                el.classList.toggle('active', i === index);
            });

            updatePrice();
        }

        function selectSpoiler(index) {
            currentSpoiler = index;
            createSpoiler();
            
            document.querySelectorAll('.spoiler-btn').forEach((el, i) => {
                el.classList.toggle('active', i === index);
            });

            updatePrice();
        }

        function setEngine(engine) {
            currentEngine = engine;
            updatePrice();
        }

        function updatePrice() {
            let customPrice = 0;
            if (currentWheel > 0) customPrice += 1500 * currentWheel;
            if (currentSpoiler > 0) customPrice += 2000 * currentSpoiler;
            if (currentEngine === 'turbo') customPrice += 5000;
            if (currentEngine === 'v8') customPrice += 12000;

            totalPrice = basePrice + customPrice;
            document.getElementById('price').textContent = `$${totalPrice.toLocaleString()}`;

            const stats = document.getElementById('stats');
            stats.innerHTML = `
                Base Price: $${basePrice.toLocaleString()}<br>
                Customizations: $${customPrice.toLocaleString()}
            `;
        }

        function saveDesign() {
            const canvas = renderer.domElement.toDataURL('image/png');
            const link = document.createElement('a');
            link.download = `custom-car-${Date.now()}.png`;
            link.href = canvas;
            link.click();

            // Show success message
            const btn = document.getElementById('saveBtn');
            const originalText = btn.textContent;
            btn.textContent = '✅ Design Saved!';
            btn.style.background = 'linear-gradient(45deg, #51cf66, #74c69d)';
            setTimeout(() => {
                btn.textContent = originalText;
                btn.style.background = 'linear-gradient(45deg, #ff6b6b, #ff8e8e)';
            }, 2000);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }

        // Handle window resize
        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth * 0.65 / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth * 0.65, window.innerHeight);
        });
    </script>
